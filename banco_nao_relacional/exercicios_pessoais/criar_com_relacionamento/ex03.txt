    {
        $lookup: {
            from: "pokemons",
            localField: "pokemons",
            foreignField: "_id",
            as: "pokes"
        }
    },
    {
        $project: {
            nome: 1, // Exibe o nome do treinador
            "pokes.nome": 1 // Exibe o nome dos pok√©mons
        }
    }
]).pretty()
