db.pokemons.aggregate([
  {
    $match: {
      $expr: {
        $gte: [{ $size: "$habilidades" }, 3]
      }
    }
  }
  ,
    {
        $project: {
            _id: 0,
            nome: 1,
            habilidades: 1
        }
    }
])
