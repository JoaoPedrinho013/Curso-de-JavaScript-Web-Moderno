7 Desenvolva uma função que receba dois números e retorne o maior deles.
8 Crie uma arrow function que calcule e retorne o fatorial de um número.

10 Desenvolva uma arrow function que receba dois valores e retorne o resto da divisão do primeiro pelo segundo.

EXEMPLO:

const operacoesMatematicas = (valor1, valor2) => {
  const soma = valor1 + valor2;
  const subtracao = valor1 - valor2;
  const multiplicacao = valor1 * valor2;
  const divisao = valor1 / valor2;

  console.log(`Soma: ${soma}`);
  console.log(`Subtração: ${subtracao}`);
  console.log(`Multiplicação: ${multiplicacao}`);
  console.log(`Divisão: ${divisao}`);
};

// Exemplo de uso
operacoesMatematicas(10, 5);
